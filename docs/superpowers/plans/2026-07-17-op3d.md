# op3d.html Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `op3d.html` — the Lab White portfolio page upgraded with interactive Three.js 3D hardware (hero gadget + one widget per module) per `docs/superpowers/specs/2026-07-17-op3d-design.md`.

**Architecture:** Copy of `index.html` content/layout with a single shared WebGL canvas (fixed, full-viewport, `pointer-events:none`) scissor-rendered into DOM placeholder rects; on-demand rendering (dirty flags + rAF only while animating). SVG art stays in the HTML as fallback; successful WebGL boot adds a `gl` class that hides it.

**Tech Stack:** Vanilla HTML/CSS/JS + Three.js `0.170.0` pinned from jsDelivr via import map. No build, no other deps, no webfonts, no external assets (all textures procedural via `CanvasTexture`).

## Global Constraints

- One new file only: `op3d.html`. `index.html`, `grid.html`, sub-projects untouched.
- Dependency: exactly `https://cdn.jsdelivr.net/npm/three@0.170.0/build/three.module.js` (import map name `three`). Nothing else external.
- Design tokens, copy, links, header/SND/QWER/readout/reveal behaviors: identical to `index.html`.
- CDN/WebGL failure must leave a fully working page with zero console errors (SVG fallbacks remain).
- No `prefers-reduced-motion` handling (explicit standing user decision — do not re-add).
- Orthographic cameras only; static clinical lighting; no cursor-tracking light; no continuous idle render loop (the CSS grain overlay is the only always-on animation).
- Materials: chassis metalness 0.9 / roughness 0.5; keys metalness 0 / roughness 0.9 in page toy colors; knobs dark low-specular rubber with ridges; screen emissive dot-matrix.
- Uppercase only in small tracked mono labels (existing rules-break exception).

### Task 2: WebGL boot, scissor pipeline, hero gadget (static)

**Files:**
- Modify: `op3d.html` (add import map + module script before `</body>`, after the existing inline script)

**Interfaces:**
- Produces (used by Tasks 3–4, all inside the same module script):
  - `views` — `Map<string elemDataViewName, View>`; `View = { elem, scene, camera, size(w,h)=>void, hit?, anims:Set }`
  - `register(name, builder)` — builder receives `(scene)` and returns the optional per-view api object stored on `view.api`
  - `invalidate()` — mark all visible views dirty and request a frame
  - `animate(fn)` — add `fn(dt)=>boolean` to the loop; return false to stop; loop self-stops when no anims and not dirty
  - `spring(get, set, target, {k=900, c=14})` — runs a stiff spring via `animate`, calls `set(value)`
  - `stdScene(scene)` — adds ambient + shadowed directional light + ground `ShadowMaterial` plane, returns nothing
  - `ALU`, `PLASTIC(color)`, `RUBBER` — shared `MeshStandardMaterial` factories
  - `screenTex` — `{ tex: CanvasTexture, print(text) }` dot-matrix canvas for the hero screen

- [ ] **Step 2: Scissor render pipeline (render-on-demand)**

```js
const views = new Map();
let dirty = false, raf = 0;
const anims = new Set();

function register(name, builder) {
  const elem = document.querySelector(`[data-view="${name}"]`);
  if (!elem) return;
  const scene = new THREE.Scene();
  const camera = new THREE.OrthographicCamera(-1, 1, 1, -1, 0.1, 100);
  camera.position.set(5, 5, 5); // isometric-ish
  camera.lookAt(0, 0.4, 0);
  const view = { elem, scene, camera, api: null };
  view.api = builder(scene, view) || {};
  views.set(name, view);
}

function frame() {
  raf = 0;
  const dt = 1 / 60;
  for (const fn of [...anims]) if (fn(dt) === false) anims.delete(fn);
  renderer.setSize(innerWidth, innerHeight, false);
  renderer.setScissorTest(false);
  renderer.clear();               // wipe stale rects from previous scroll positions
  renderer.setScissorTest(true);
  for (const { elem, scene, camera } of views.values()) {
    const r = elem.getBoundingClientRect();
    if (r.bottom < 0 || r.top > innerHeight || r.right < 0 || r.left > innerWidth) continue;
    const zoom = 1.6; // world units of half-height
    camera.left = -zoom * (r.width / r.height); camera.right = zoom * (r.width / r.height);
    camera.top = zoom; camera.bottom = -zoom;
    camera.updateProjectionMatrix();
    const y = innerHeight - r.bottom;
    renderer.setViewport(r.left, y, r.width, r.height);
    renderer.setScissor(r.left, y, r.width, r.height);
    renderer.render(scene, camera);
  }
  dirty = false;
  if (anims.size) invalidate();
}
function invalidate() { dirty = true; if (!raf) raf = requestAnimationFrame(frame); }
function animate(fn) { anims.add(fn); invalidate(); }
addEventListener('scroll', invalidate, { passive: true });
addEventListener('resize', invalidate);
```

- [ ] **Step 3: Shared materials, lighting, spring**

```js
const ALU = () => new THREE.MeshStandardMaterial({ color: 0xdcdcda, metalness: 0.9, roughness: 0.5 });
const PLASTIC = (c) => new THREE.MeshStandardMaterial({ color: c, metalness: 0, roughness: 0.9 });
const RUBBER = () => new THREE.MeshStandardMaterial({ color: 0x222222, metalness: 0, roughness: 0.85 });
const TOY = { orange: 0xff4d00, cyan: 0x00a4c8, yellow: 0xf5c400, green: 0x3fa435 };

function stdScene(scene) {
  scene.add(new THREE.AmbientLight(0xffffff, 1.1));
  const sun = new THREE.DirectionalLight(0xffffff, 2.2);
  sun.position.set(2, 8, 3);
  sun.castShadow = true;
  sun.shadow.mapSize.set(1024, 1024);
  scene.add(sun);
  const ground = new THREE.Mesh(new THREE.PlaneGeometry(20, 20), new THREE.ShadowMaterial({ opacity: 0.18 }));
  ground.rotation.x = -Math.PI / 2;
  ground.receiveShadow = true;
  scene.add(ground);
}

function spring(get, set, target, { k = 900, c = 14 } = {}) {
  let v = 0;
  animate((dt) => {
    let x = get();
    const a = -k * (x - target) - c * v;
    v += a * dt; x += v * dt;
    if (Math.abs(x - target) < 0.001 && Math.abs(v) < 0.01) { set(target); return false; }
    set(x); return true;
  });
}
```

- [ ] **Step 1: Raycast helper + key press**

```js
const ray = new THREE.Raycaster(), ndc = new THREE.Vector2();
function pick(view, e, objects) {
  const r = view.elem.getBoundingClientRect();
  ndc.set(((e.clientX - r.left) / r.width) * 2 - 1, -(((e.clientY - r.top) / r.height) * 2 - 1));
  ray.setFromCamera(ndc, view.camera);
  return ray.intersectObjects(objects, true)[0]?.object ?? null;
}
```

Hero elem `pointerdown`: find pressed key (walk up `.parent` until `userData.rest` found). On hit: `click()`; spring key to `rest - 0.12` then, on a 90 ms timeout, spring back to `rest` with default stiff params (sharp recoil); `modules[i].scrollIntoView({behavior:'smooth', block:'start'})` reusing the page's `modules` array (module script reads it via `['work','lab','log','spec'].map(id => document.getElementById(id))` — same lookup, no coupling).

- [ ] **Step 2: Detent engine**

```js
const DETENT = Math.PI / 12; // 15 degrees
function makeDetent(object, onStep) {
  let acc = 0, step = 0;
  function add(delta) {           // delta: radians of raw input
    acc += delta;
    const s = Math.round(acc / DETENT);
    if (s !== step) {
      const dir = Math.sign(s - step); step = s;
      click();
      onStep?.(dir, s);
      // snap with micro-shudder: overshoot then settle
      spring(() => object.rotation.y, (v) => { object.rotation.y = v; invalidate(); }, s * DETENT, { k: 1400, c: 10 });
    }
  }
  return { add };
}
```
